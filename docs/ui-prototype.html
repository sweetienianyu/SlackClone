分享一下 WebSocket 重连的核心代码逻辑：
class SocketManager {
  private reconnectAttempts = 0;
  private maxReconnectAttempts = 10;

  async connect() {
    this.socket = io(SOCKET_URL, {
      auth: { token: this.token },
      reconnection: true,
      reconnectionDelay: this.getBackoffDelay(),
    });
    this.socket.on('disconnect', () => this.enterOfflineMode());
    this.socket.on('reconnect', async () => {
      await this.syncDeltaMessages();
      this.exitOfflineMode();
    });
  }

  private getBackoffDelay(): number {
    return Math.min(1000 * Math.pow(2, this.reconnectAttempts), 30000);
  }
}
